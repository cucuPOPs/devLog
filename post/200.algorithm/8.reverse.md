## 단어단위로 뒤집기

```Python
def reverse_words(str):
    words=str.split(" ")
    words.reverse()
    return " ".join(words)

def reverse_words2(str):
    words = str.split(" ")
    return " ".join(reversed(words))

def reverse_words3(str):
    new_words = []
    words = str.split(" ")
    for i in words[::-1]:
        new_words.append(i)
    return " ".join(new_words)
```
