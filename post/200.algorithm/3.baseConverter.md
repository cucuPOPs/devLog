## 다른 진법을 10진수로

```Python
def convertToDecimal(number, base):
    multiplier, result = 1, 0
    while number > 0:
        result += number % 10 * multiplier
        multiplier *= base
        number = number // 10
    return result

def test_convertToDecimal():
    number, base = 1001, 2
    assert convertToDecimal(number, base) == 9
    print("테스트통과!")

```

## 10진법을 다른진법으로

```Python
def convertFromDecimal(number, base):
    multiplier, result = 1, 0
    while number > 0:
        result += number % base * multiplier
        multiplier *= 10
        number = number // base
    return result

def test_convertFromDecimal():
    number, base = 9, 2
    assert convertToDecimal(number, base) == 1001
    print("테스트통과!")
```

## 10진법을 10이상진법으로

```Python

## 10진수를 20진법으로
def convertFromDecimalLargerBase(number, base):
    strings = "0123456789ABCDEFGHIJ"
    result = ""
    while number > 0:
        digit = number % base
        result = strings[digit] + result
        number = number // base
    return result

def test_convertFromDecimalLargerBase():
    number, base = 31, 16
    assert convertFromDecimalLargerBase(number, base) == "1F"
    print("테스트통과!")
```

## 재귀함수로 진법변환

```Python
def convertToAnyBase(number, base):
    convertString = "0123456789ABCDEF"
    if number < base:
        return convertString[number]
    else:
        return convertToAnyBase(number // base, base) + convertString[number % base]

def test_convertToAnyBase():
    number = 9
    base = 2
    assert(convertToAnyBase(number,base)=="1001")
    print("테스트통과!")
```
