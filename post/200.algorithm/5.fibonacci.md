## 피보나치 수열
def fibonacciSeqIter(n):
    if n < 2:
        return n
    a, b = 0, 1
    for _ in range(n):
        a, b = b, a + b
    return a

def fibonacciSeqRec(n):
    if n < 2:
        return n
    return fibonacciSeqRec(n - 1) + fibonacciSeqRec(n - 2)

def fib_generator():
    a, b = 0, 1
    while True:
        yield b
        a, b = b, a + b

def testfib_generator():
    fg = fib_generator()
    for _ in range(10):
        print(next(fg), end=" ")
