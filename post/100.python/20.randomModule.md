```Python
import random

def test_random():
    myList=[1,2,3,4]
    print(random.choice(myList)) # 1개 랜덤으로pick
    print(random.sample(myList,2)) # 2개 pick(중복X) 리스트로 반환
    random.shuffle(myList) # 섞는다

    print(random.randint(0,10))# 0포함~10포함
```
